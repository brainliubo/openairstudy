76 #define RLC_AM_MUTEX_LOCK(mUTEX, cTXT, rLC) \
77  do {\
78  int pmtl_rc = pthread_mutex_trylock(mUTEX);\
79  if (pmtl_rc != 0){\
80  if (pmtl_rc == EBUSY) {\
81  MSC_LOG_EVENT((cTXT->enb_flag == ENB_FLAG_YES) ? MSC_RLC_ENB:MSC_RLC_UE,\
82  "0 "PROTOCOL_RLC_AM_MSC_FMT" Warning try lock %s busy",\
83  PROTOCOL_RLC_AM_MSC_ARGS(cTXT,rLC),\
84  #mUTEX);\
85  pthread_mutex_lock(mUTEX);\
86  } else {\
87  MSC_LOG_EVENT((cTXT->enb_flag == ENB_FLAG_YES) ? MSC_RLC_ENB:MSC_RLC_UE,\
88  "0 "PROTOCOL_RLC_AM_MSC_FMT" Error try lock %s %d",\
89  PROTOCOL_RLC_AM_MSC_ARGS(cTXT,rLC),\
90  #mUTEX, pmtl_rc);\
91  }\
92  }\
93  } while (0);
