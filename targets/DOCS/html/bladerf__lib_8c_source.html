183 static int trx_brf_read(openair0_device *device, openair0_timestamp *ptimestamp, void **buff, int nsamps, int cc) {
184 
185  int status=0;

188  // BRF has only one rx/tx chain
189  int16_t *samples = (int16_t*)buff[0];

191  brf->meta_rx.actual_count = 0;
192  brf->meta_rx.flags = BLADERF_META_FLAG_RX_NOW;
193  status = bladerf_sync_rx(brf->dev, samples, (unsigned int) nsamps, &brf->meta_rx, 2*brf->rx_timeout_ms);
194 
195  // printf("Current RX timestampe %u, nsamps %u, actual %u, cc %d\n", brf->meta_rx.timestamp, nsamps, brf->meta_rx.actual_count, cc);
196 
197  if (status != 0) {
198  fprintf(stderr, "RX failed: %s\n", bladerf_strerror(status));
199  // printf("RX failed: %s\n", bladerf_strerror(status));
200  brf->num_rx_errors++;
201  } else if ( brf->meta_rx.status & BLADERF_META_STATUS_OVERRUN) {
202  brf->num_overflows++;
203  printf("RX overrun (%d) is detected. t=" "%" PRIu64 "Got %u samples. nsymps %d\n",
204  brf->num_overflows,brf->meta_rx.timestamp, brf->meta_rx.actual_count, nsamps);
205  }
206 
207  if (brf->meta_rx.actual_count != nsamps) {
208  printf("RX bad samples count, wanted %d, got %d\n", nsamps, brf->meta_rx.actual_count);
209  }
210 
211  //printf("Current RX timestampe %u\n", brf->meta_rx.timestamp);
212  //printf("[BRF] (buff %p) ts=0x%"PRIu64" %s\n",samples, brf->meta_rx.timestamp,bladerf_strerror(status));
213  brf->rx_current_ts=brf->meta_rx.timestamp;
214  brf->rx_actual_nsamps+=brf->meta_rx.actual_count;
215  brf->rx_nsamps+=nsamps;
216  brf->rx_count++;
217 
218 
219  *ptimestamp = brf->meta_rx.timestamp;
220 
221  return nsamps; //brf->meta_rx.actual_count;
