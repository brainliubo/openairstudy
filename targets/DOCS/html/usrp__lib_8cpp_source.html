302 static int trx_usrp_start(openair0_device *device) {
303 #if defined(USRP_REC_PLAY)
304 
305  if (u_sf_mode != 2) { // not replay mode
306 #endif
307  usrp_state_t *s = (usrp_state_t *)device->priv;
308  // setup GPIO for TDD, GPIO(4) = ATR_RX
309  //set data direction register (DDR) to output
310  s->usrp->set_gpio_attr("FP0", "DDR", 0x1f, 0x1f);
311  //set control register to ATR
312  s->usrp->set_gpio_attr("FP0", "CTRL", 0x1f,0x1f);
313  //set ATR register
314  s->usrp->set_gpio_attr("FP0", "ATR_RX", 1<<4, 0x1f);
315  // init recv and send streaming
316  uhd::stream_cmd_t cmd(uhd::stream_cmd_t::STREAM_MODE_START_CONTINUOUS);
317  LOG_I(PHY,"Time in secs now: %llu \n", s->usrp->get_time_now().to_ticks(s->sample_rate));
318  LOG_I(PHY,"Time in secs last pps: %llu \n", s->usrp->get_time_last_pps().to_ticks(s->sample_rate));
319 
320  if (s->use_gps == 1) {
321  s->wait_for_first_pps = 1;
322  cmd.time_spec = s->usrp->get_time_last_pps() + uhd::time_spec_t(1.0);
323  } else {
324  s->wait_for_first_pps = 0;
325  cmd.time_spec = s->usrp->get_time_now() + uhd::time_spec_t(0.05);
326  }
327 
328  cmd.stream_now = false; // start at constant delay
329  s->rx_stream->issue_stream_cmd(cmd);
330  s->tx_md.time_spec = cmd.time_spec + uhd::time_spec_t(1-(double)s->tx_forward_nsamps/s->sample_rate);
331  s->tx_md.has_time_spec = true;
332  s->tx_md.start_of_burst = true;
333  s->tx_md.end_of_burst = false;
334  s->rx_count = 0;
335  s->tx_count = 0;
336  s->rx_timestamp = 0;
337 #if defined(USRP_REC_PLAY)
338  }
339 
340 #endif
341  return 0;

435 static int trx_usrp_write(openair0_device *device, openair0_timestamp timestamp, void **buff, int nsamps, int cc, int flags) {
436  int ret=0;
437 #if defined(USRP_REC_PLAY)
438 
439  if (u_sf_mode != 2) { // not replay mode
440 #endif
441  usrp_state_t *s = (usrp_state_t *)device->priv;
442  int nsamps2; // aligned to upper 32 or 16 byte boundary
443 #if defined(__x86_64) || defined(__i386__)
444 #ifdef __AVX2__
445  nsamps2 = (nsamps+7)>>3;
446  __m256i buff_tx[2][nsamps2];
447 #else
448  nsamps2 = (nsamps+3)>>2;
449  __m128i buff_tx[2][nsamps2];
450 #endif
451 #elif defined(__arm__)
452  nsamps2 = (nsamps+3)>>2;
453  int16x8_t buff_tx[2][nsamps2];
454 #endif
455 
456  // bring RX data into 12 LSBs for softmodem RX
457  for (int i=0; i<cc; i++) {
458  for (int j=0; j<nsamps2; j++) {
459 #if defined(__x86_64__) || defined(__i386__)
460 #ifdef __AVX2__
461  buff_tx[i][j] = _mm256_slli_epi16(((__m256i *)buff[i])[j],4);
462 #else
463  buff_tx[i][j] = _mm_slli_epi16(((__m128i *)buff[i])[j],4);
464 #endif
465 #elif defined(__arm__)
466  buff_tx[i][j] = vshlq_n_s16(((int16x8_t *)buff[i])[j],4);
467 #endif
468  }
469  }
470 
471  s->tx_md.time_spec = uhd::time_spec_t::from_ticks(timestamp, s->sample_rate);
472  s->tx_md.has_time_spec = flags;
473 
474  if(flags>0)
475  s->tx_md.has_time_spec = true;
476  else
477  s->tx_md.has_time_spec = false;
478 
479  if (flags == 2) { // start of burst
480  s->tx_md.start_of_burst = true;
481  s->tx_md.end_of_burst = false;
482  } else if (flags == 3) { // end of burst
483  s->tx_md.start_of_burst = false;
484  s->tx_md.end_of_burst = true;
485  } else if (flags == 4) { // start and end
486  s->tx_md.start_of_burst = true;
487  s->tx_md.end_of_burst = true;
488  } else if (flags==1) { // middle of burst
489  s->tx_md.start_of_burst = false;
490  s->tx_md.end_of_burst = false;
491  }
492 
493  if(flags==10) { // fail safe mode
494  s->tx_md.has_time_spec = false;
495  s->tx_md.start_of_burst = false;
496  s->tx_md.end_of_burst = true;
497  }
498 
499  if (cc>1) {
500  std::vector<void *> buff_ptrs;
501 
502  for (int i=0; i<cc; i++)
503  buff_ptrs.push_back(buff_tx[i]);
504 
505  ret = (int)s->tx_stream->send(buff_ptrs, nsamps, s->tx_md,1e-3);
506  } else
507  ret = (int)s->tx_stream->send(buff_tx[0], nsamps, s->tx_md,1e-3);
508 
509  if (ret != nsamps)
510  LOG_E(PHY,"[xmit] tx samples %d != %d\n",ret,nsamps);
511 
512 #if defined(USRP_REC_PLAY)
513  } else {
514  struct timespec req;
515  req.tv_sec = 0;
516  req.tv_nsec = u_sf_write_delay * 1000;
517  nanosleep(&req, NULL);
518  ret = nsamps;
519  }
520 
521 #endif
522  return ret;
523 }

536 static int trx_usrp_read(openair0_device *device, openair0_timestamp *ptimestamp, void **buff, int nsamps, int cc) {
537  usrp_state_t *s = (usrp_state_t *)device->priv;
538  int samples_received=0,i,j;
539  int nsamps2; // aligned to upper 32 or 16 byte boundary
540 #if defined(USRP_REC_PLAY)
541 
542  if (u_sf_mode != 2) { // not replay mode
543 #endif
544 #if defined(__x86_64) || defined(__i386__)
545 #ifdef __AVX2__
546  nsamps2 = (nsamps+7)>>3;
547  __m256i buff_tmp[2][nsamps2];
548 #else
549  nsamps2 = (nsamps+3)>>2;
550  __m128i buff_tmp[2][nsamps2];
551 #endif
552 #elif defined(__arm__)
553  nsamps2 = (nsamps+3)>>2;
554  int16x8_t buff_tmp[2][nsamps2];
555 #endif
556 
557  if (device->type == USRP_B200_DEV) {
558  if (cc>1) {
559  // receive multiple channels (e.g. RF A and RF B)
560  std::vector<void *> buff_ptrs;
561 
562  for (int i=0; i<cc; i++) buff_ptrs.push_back(buff_tmp[i]);
563 
564  samples_received = s->rx_stream->recv(buff_ptrs, nsamps, s->rx_md);
565  } else {
566  // receive a single channel (e.g. from connector RF A)
567  samples_received=0;
568 
569  while (samples_received != nsamps) {
570  samples_received += s->rx_stream->recv(buff_tmp[0]+samples_received,
571  nsamps-samples_received, s->rx_md);
572 
573  if ((s->wait_for_first_pps == 0) && (s->rx_md.error_code!=uhd::rx_metadata_t::ERROR_CODE_NONE))
574  break;
575 
576  if ((s->wait_for_first_pps == 1) && (samples_received != nsamps)) {
577  printf("sleep...\n"); //usleep(100);
578  }
579  }
580 
581  if (samples_received == nsamps) s->wait_for_first_pps=0;
582  }
583 
584  // bring RX data into 12 LSBs for softmodem RX
585  for (int i=0; i<cc; i++) {
586  for (int j=0; j<nsamps2; j++) {
587 #if defined(__x86_64__) || defined(__i386__)
588 #ifdef __AVX2__
589  ((__m256i *)buff[i])[j] = _mm256_srai_epi16(buff_tmp[i][j],4);
590 #else
591  ((__m128i *)buff[i])[j] = _mm_srai_epi16(buff_tmp[i][j],4);
592 #endif
593 #elif defined(__arm__)
